CREATE TRIGGER [dbo].[increaseStock] 
ON ILAC 
FOR DELETE
AS 
BEGIN
UPDATE ILAC SET STOK=STOK+1 WHERE ILACID=(SELECT ILACID FROM DELETED)
END


CREATE TRIGGER [dbo].[HASTAILAC] 
ON  [dbo].HASTAILAC 
FOR INSERT
AS 
BEGIN
UPDATE ILAC SET STOK=STOK-1 WHERE ILACID=(SELECT ILACID FROM INSERTED)
END
